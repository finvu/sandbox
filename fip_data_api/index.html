    FIP Data API

    This API can be used to add sample data for accounts and associated transactions and can be called from an appropriate testing tool such as Postman

    Adding User Account
    URL http://api.finvu.in/Accounts/add
    HTTP METHOD POST

    Request Body

    <UserAccountInfo>
    <UserAccount accountRefNo="RefHDFC0001" accountNo="HDFC11111111" accountTypeEnum="SAVINGS" FIType="DEPOSIT">
      <Identifiers pan="AFYPA24444" mobile="7030608902" email="info@cookiejar.co.in" aadhar="679486011111"></Identifiers> 
    </UserAccount>
    <UserAccount accountRefNo="RefHDFC0002" accountNo="HDFC22222222" accountTypeEnum="CURRENT" FIType="DEPOSIT">
      <Identifiers pan="AFYPA24444" mobile="7030608902" email="info@cookiejar.co.in" aadhar="679486011111"></Identifiers> 
    </UserAccount>
    <UserAccount accountRefNo="RefSBI0001" accountNo="SBI11111111" accountTypeEnum="SAVINGS" FIType="DEPOSIT">
      <Identifiers pan="AFYPA24444" mobile="7030608902" email="info@cookiejar.co.in" aadhar="679486011111"></Identifiers> 
    </UserAccount>
</UserAccountInfo>
